#include<bits/stdc++.h>
#define N 50010
#define ll long long
using namespace std;
ll t,n,m,p[N+1],s[N+1],mo[N+1],g[N+1];
bool used[N+1];
ll gg(ll x)
{
    ll ans=0;
    for(ll i=1,j;i<=x;i=j+1)
    {
        j=x/(x/i);
        ans+=(j-i+1)*(x/i);
    }
    return ans;
}
void euler()
{
    mo[1]=1;
    for(ll i=2;i<N;i++)
    {
        if(used[i]==0)
        {
            p[++p[0]]=i;
            mo[i]=-1;
            used[i]=1;
        }
        for(int j=1;j<=p[0];j++)
        {
            if(i*p[j]>=N)
                break;
            used[i*p[j]]=1;
            if(i%p[j]==0)
            {
                mo[i*p[j]]=0;
                break;
            }
            mo[i*p[j]]=-mo[i];
        }
    }
    for(ll i=1;i<N;i++)
        s[i]=s[i-1]+mo[i];
}
int main()
{
    scanf("%lld",&t);
    euler();
    for(int i=1;i<=N;i++)
        g[i]=gg(i);
    while(t--)
    {
        scanf("%lld%lld",&n,&m);
        ll ans=0;
        for(ll i=1,j;i<=min(n,m);i=j+1)
        {
            j=min(n/(n/i),m/(m/i));
            ans+=g[n/i]*g[m/i]*(s[j]-s[i-1]);
        }
        printf("%lld\n",ans);
    }
    return 0;
}
